00230   {

00233     TemplateTypeInterface* futureType = ft1 ? ft1 : ft2;
00234     qiLogDebug("qi.object") << "isFuture " << val.type()->infoString() << ' ' << !!ft1 << ' ' << !!ft2;
00235     if (!futureType)
00236       return false;
00237 
00238     TypeInterface* next = futureType->next();
