00168 template<typename ProxyType>
00169 bool registerProxy()
00170 {
00171   /* In non-interface mode, we need one different registered type
00172    * per Proxy class, otherwise they will all land in the same
00173    * bucket of proxyGeneratorMap.
00174    * In interface mode this problem is absent because the TypeInfo
00175    * of the interface is used (not the one of the proxy).
00176   */
00177   registerType(typeid(ProxyType), detail::makeProxyInterfaceWrapper<ProxyType>());
00178   detail::ProxyGeneratorMap& map = detail::proxyGeneratorMap();
00179   map[typeOf<ProxyType>()->info()] = &detail::makeProxy<ProxyType>;
00180   return true;
